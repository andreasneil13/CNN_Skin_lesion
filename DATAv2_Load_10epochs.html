import pandas as pd
import os

# Dossier d'images et fichier de vérité terrain
images_dir = "ISIC2019/ISIC_2019_Training_Input/ISIC_2019_Training_Input"
metadata_path = "ISIC2019/ISIC_2019_Training_GroundTruth.csv"

# Lire le CSV
df = pd.read_csv(metadata_path)
print("Nombre total d'images (brut) :", len(df))

# Colonnes d'intérêt
label_cols = ["MEL","NV","BCC","AK","BKL","DF","VASC","SCC"]

# Mapper les colonnes one-hot vers 7 labels
label_map = {
    "NV": "nv",
    "MEL": "mel",
    "BKL": "bkl",
    "BCC": "bcc",
    "AK": "akiec",   # fusion AK
    "SCC": "akiec",  # fusion SCC
    "VASC": "vasc",
    "DF": "df"
}

def one_hot_to_label(row):
    positive = row[label_cols][row[label_cols] == 1].index.tolist()
    if len(positive) != 1:
        return None
    return label_map[positive[0]]

# Générer les colonnes path et label
df["label"] = df.apply(one_hot_to_label, axis=1)
df["path"] = df["image"].apply(lambda x: os.path.join(images_dir, f"{x}.jpg"))

# Supprimer les lignes sans label (ex: multi-labels rares)
df = df.dropna(subset=["label"])

# Vérifier les images manquantes
df = df[df["path"].apply(os.path.exists)]
print("Images valides après vérif :", len(df))
print(df[["path", "label"]].head())

import tensorflow.keras as keras
from tensorflow.keras import layers
from tensorflow.keras.applications import EfficientNetB0

def build_efficientnet_b0(img_size: int = 224,
                          num_classes: int = NUM_CLASSES,
                          dropout_rate: float = 0.5,
                          base_trainable: bool = False) -> keras.Model:
    """
    Construit un modèle EfficientNetB0 + tête de classification.
    Args:
        img_size : côté de l'image carrée en pixels.
        num_classes : nombre de classes en sortie.
        dropout_rate : taux de dropout avant la Dense finale.
        base_trainable : si True, EfficientNet n'est pas gelé (fine‑tuning).
    Returns:
        keras.Model compilé (non encore entraîné).
    """
    base_model = EfficientNetB0(
        weights="imagenet",
        include_top=False,
        input_shape=(img_size, img_size, 3)
    )
    base_model.trainable = base_trainable   # gèle ou non le backbone

    inputs = keras.Input(shape=(img_size, img_size, 3), name="input_rgb")
    x = base_model(inputs, training=False)          # invariance batch‑norm
    x = layers.GlobalAveragePooling2D(name="gap")(x)
    x = layers.Dropout(dropout_rate, name="dropout")(x)
    outputs = layers.Dense(num_classes, activation="softmax", dtype="float32",
                           name="predictions")(x)   # force sortie float32
    return keras.Model(inputs, outputs, name="EffNetB0_skin")

import matplotlib.pyplot as plt

def plot_training_curves(history_list, metric="accuracy"):
    """
    Affiche sur le même graphe les courbes train/val d'une métrique
    pour un ou plusieurs objets `keras.callbacks.History`.
    """
    if not isinstance(history_list, (list, tuple)):
        history_list = [history_list]

    plt.figure(figsize=(6,4))
    for idx, hist in enumerate(history_list):
        label_train = f"train – run {idx+1}"
        label_val   = f"val   – run {idx+1}"
        plt.plot(hist.history[metric],        lw=2, label=label_train)
        plt.plot(hist.history[f"val_{metric}"], lw=2, label=label_val)

    plt.title(f"Évolution de {metric}")
    plt.xlabel("Epoch")
    plt.ylabel(metric.capitalize())
    plt.legend()
    plt.grid(True)
    plt.tight_layout()
    plt.show()

from sklearn.preprocessing import label_binarize
from sklearn.metrics import roc_curve, auc
import matplotlib.pyplot as plt
from itertools import cycle

# Y_true en one‑hot
y_true_bin = label_binarize(y_true, classes=np.arange(NUM_CLASSES))

fpr = dict()
tpr = dict()
roc_auc = dict()

for i in range(NUM_CLASSES):
    fpr[i], tpr[i], _ = roc_curve(y_true_bin[:, i], y_proba[:, i])
    roc_auc[i] = auc(fpr[i], tpr[i])

# Micro‑average et macro‑average
fpr["micro"], tpr["micro"], _ = roc_curve(y_true_bin.ravel(), y_proba.ravel())
roc_auc["micro"] = auc(fpr["micro"], tpr["micro"])

# Agrégation macro
all_fpr = np.unique(np.concatenate([fpr[i] for i in range(NUM_CLASSES)]))
mean_tpr = np.zeros_like(all_fpr)
for i in range(NUM_CLASSES):

    mean_tpr += np.interp(all_fpr, fpr[i], tpr[i])
mean_tpr /= NUM_CLASSES
fpr["macro"], tpr["macro"], roc_auc["macro"] = all_fpr, mean_tpr, auc(all_fpr, mean_tpr)

# Affichage
plt.figure(figsize=(7,6))
colors = cycle(plt.cm.tab10(np.linspace(0, 1, NUM_CLASSES)))

for i, color in zip(range(NUM_CLASSES), colors):
    plt.plot(fpr[i], tpr[i], color=color, lw=1.5,
             label=f"{class_names[i]} (AUC ={roc_auc[i]:.2f})")

plt.plot(fpr["micro"], tpr["micro"], linestyle="--", color="grey",
         label=f"micro‑avg (AUC ={roc_auc['micro']:.2f})", lw=2)
plt.plot(fpr["macro"], tpr["macro"], linestyle=":", color="black",
         label=f"macro‑avg (AUC ={roc_auc['macro']:.2f})", lw=2)
