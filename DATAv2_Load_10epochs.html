import tensorflow.keras as keras
from tensorflow.keras import layers
from tensorflow.keras.applications import EfficientNetB0

def build_efficientnet_b0(img_size: int = 224,
                          num_classes: int = NUM_CLASSES,
                          dropout_rate: float = 0.5,
                          base_trainable: bool = False) -> keras.Model:
    """
    Construit un modle EfficientNetB0 + tte de classification.
    Args:
        img_size : ct de l'image carre en pixels.
        num_classes : nombre de classes en sortie.
        dropout_rate : taux de dropout avant la Dense finale.
        base_trainable : si True, EfficientNet n'est pas gel (finetuning).
    Returns:
        keras.Model compil (non encore entran).
    """
    base_model = EfficientNetB0(
        weights="imagenet",
        include_top=False,
        input_shape=(img_size, img_size, 3)
    )
    base_model.trainable = base_trainable   # gle ou non le backbone

    inputs = keras.Input(shape=(img_size, img_size, 3), name="input_rgb")
    x = base_model(inputs, training=False)          # invariance batchnorm
    x = layers.GlobalAveragePooling2D(name="gap")(x)
    x = layers.Dropout(dropout_rate, name="dropout")(x)
    outputs = layers.Dense(num_classes, activation="softmax", dtype="float32",
                           name="predictions")(x)   # force sortie float32
    return keras.Model(inputs, outputs, name="EffNetB0_skin")

def plot_training_curves(history_list, metric="accuracy"):
